### **一、账号管理**



#### **1.  管理员(admin)**

##### ① 按筛选条件获取管理员列表

```java
url: '/account/admin/list'
method: 'get'
params: {
    id: '',             //根据id查询，为空时不过滤，字符串!!! 
    username: '',       //根据用户名查询，为空时不过滤
    page: 1,            //页数
    limit: 20,          //每页个数
    sort: '+id'         //排序，为'-id'时逆序
}
response: {
    list: {         //返回对象数组，根据limit确认数组大小，默认每页20个，list[20]
        { id: 123, username: 'test1', password: 'test1'},
        { id: 456, username: 'test2', password: 'test2'}
    }
    total: 200      //符合条件的总记录数
}
```

##### ② 添加管理员

```java
url: '/account/admin/create'
method: 'post'
params: {
    username: 'test',           //用户名
    password: 'test',           //密码
    comfirm_password: 'test'    //确认密码
}
response: {
    result: 0   //信号量  成功:0   用户名已存在:1   两次密码不一致:2   读写错误:3
}
```

##### ③ 修改管理员密码

```java
url: '/account/admin/update'
method: 'post'
params: {
    id: '123',                  //管理员id
    password: 'test',           //密码
    comfirm_password: 'test'    //确认密码
}
response: {
    result: 0	//信号量  成功:0   两次密码不一致:1   读写错误:2
}
```

##### ④ 删除管理员

```java
url: '/account/admin/delete'
method: 'post'
params: {
    id: '123'   //管理员id
}
response: {
    result: 0	//信号量  成功:0   失败:1
}
```

#### **3. 住户(resident)**

##### ① 按筛选条件获取住户列表

```java
url: '/account/resident/list'
method: 'get'
params: {
    id: '',             //根据id查询，为空时不过滤，字符串!!! 
    name: '',           //根据姓名查询，为空时不过滤
    locked_query: '',   /*
                        根据锁定状态查询，为空时不过滤，字符串!!! 
                        为'locked' 过滤 is_locked==true
                        为'unlocked' 过滤 is_locked==false
                        */
    health_query: '',   /*
                        根据健康状况查询，为空时不过滤，字符串!!! 
                        为'healthy' 过滤 health_status < 预警值
                        为'unhealthy' 过滤 health_status >= 预警值
                        */
    page: 1,            //页数
    limit: 20,          //每页个数
    sort: '+id'         //排序，为'-id'时逆序
}
response: {
    list: {         //返回对象数组，根据limit确认数组大小，默认每页20个，list[20]
        { 
            id: 123,
            open_id: 'test1',
            username: 'test1',
            name: 'test1',
            sex: 0,
            identity_card: '440181199907100010',
            photo: 'xxx/xxx.jpg',
            house_no: 'A01',
            health_status: 3,
            access_times: 50,
            is_locked:	false
        },
        {
            ...
        }
    }
    total: 200      //符合条件的总记录数
}
```

### 二、出入记录



#### ① 按筛选条件获取出入记录列表

```java
url: '/record/list'
method: 'get'
params: {
    id: '',             //根据记录id查询，为空时不过滤，字符串!!! 
    resident_id: '',    //根据住户id查询，为空时不过滤，字符串!!! 
    inspector_id: '',   //根据检查员id查询，为空时不过滤，字符串!!!
    date_query: '',     /*
                        根据日期查询，为空时不过滤
                        日期时间戳数组date_query[2]
                        date_query[0]为开始日期，date_query[1]为结束日期
                        */
    health_query: '',   /*
                        根据体温查询，为空时不过滤，字符串!!! 
                        为'healthy' 过滤 temperature < 发烧温度
                        为'unhealthy' 过滤 temperature >= 发烧温度
                        */
    page: 1,            //页数
    limit: 20,          //每页个数
    sort: '+id'         //排序，为'-id'时逆序
}
response: {
    list: {         //返回对象数组，根据limit确认数组大小，默认每页20个，list[20]
        { 
            id: 123,                //记录id
            time: 90997278243,      //日期时间戳
            temperature: 37.5,      //体温
            resident_id: 123,       //住户id
            inspector_id: 123       //检查员id
        },
        {
            ...
        }
    }
    total: 200      //符合条件的总记录数
}
```
